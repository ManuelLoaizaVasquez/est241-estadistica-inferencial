\begin{statement}{18}
  La brecha financiera de un pa\'is $X \sim \text{Lognormal}(\mu,\, \sigma^2 = 4)$.
\end{statement}

\begin{statement}{a}
  Halle la probabilidad de que la brecha resulte mayor de lo esperado.
\end{statement}

\begin{statement}{b}
  Se sabe que el pa\'is tendr\'a que devaluar si la brecha pasa de cierto
  punto $x_0$. Si $\mu = 1$, ¿qu\'e valor $x_0$ hace que sea igual de probable
  devaluar o no devaluar?
\end{statement}
